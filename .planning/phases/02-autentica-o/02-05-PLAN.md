1. Create `frontend/src/components/ProtectedRoute.tsx`:

```typescript
import { Navigate, Outlet } from 'react-router-dom'
import { useAuth } from '../contexts/AuthContext'

interface Props {
  allowedRole: 'admin' | 'professor' | 'responsavel'
}

export function ProtectedRoute({ allowedRole }: Props) {
  const { isAuthenticated, user } = useAuth()

  // Not authenticated at all → go to login
  if (!isAuthenticated) {
    return <Navigate to="/login" replace />
  }

  // Authenticated but wrong role → redirect to own dashboard
  if (user?.tipo !== allowedRole) {
    return <Navigate to={`/${user?.tipo}`} replace />
  }

  return <Outlet />
}
```

Route structure:
```
/login                     → LoginPage (public; redirect to /{tipo} if already authenticated)
/esqueci-senha             → ForgotPasswordPage (public)
/redefinir-senha           → ResetPasswordPage (public; reads ?token= from URL)
/admin                     → ProtectedRoute(allowedRole="admin") → AppLayout → AdminDashboard
/professor                 → ProtectedRoute(allowedRole="professor") → AppLayout → ProfessorDashboard
/responsavel               → ProtectedRoute(allowedRole="responsavel") → AppLayout → ResponsavelDashboard
/ (root index)             → Navigate to /login (or to /{tipo} if authenticated)
* (catch-all)              → Navigate to /login
```

For /login: add a wrapper component `PublicRoute` that redirects authenticated users to `/{user.tipo}`:
```typescript
function PublicRoute({ children }: { children: React.ReactNode }) {
  const { isAuthenticated, user } = useAuth()
  if (isAuthenticated && user) {
    return <Navigate to={`/${user.tipo}`} replace />
  }
  return <>{children}</>
}
```

`RootRedirect` component:
```typescript
function RootRedirect() {
  const { isAuthenticated, user } = useAuth()
  if (isAuthenticated && user) return <Navigate to={`/${user.tipo}`} replace />
  return <Navigate to="/login" replace />
}
```

App component:
```typescript
export default function App() {
  return (
    <AuthProvider>
      <RouterProvider router={router} />
    </AuthProvider>
  )
}
```

IMPORTANT: PublicRoute and RootRedirect use useAuth(), which requires being inside AuthProvider. The router is created outside AuthProvider, so these components must be rendered inside AuthProvider via RouterProvider. This pattern works because useAuth() is called when the component renders (inside the tree), not when the router is created.
