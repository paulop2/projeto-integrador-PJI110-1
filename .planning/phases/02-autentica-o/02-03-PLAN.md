```typescript
  id: number
  email: string
  tipo: 'admin' | 'professor' | 'responsavel'
  nome: string
}
```

AuthContext value shape:
```typescript
interface AuthContextValue {
  user: AuthUser | null
  token: string | null
  login: (token: string, user: AuthUser) => void
  logout: () => void
  isAuthenticated: boolean
  updateToken: (newToken: string) => void  // Called by api.ts interceptor on X-New-Token
}
```

Implementation notes:
- Initialize state from localStorage using lazy initializer (avoids stale data on refresh):
  ```typescript
  const [user, setUser] = useState<AuthUser | null>(() => {
    const stored = localStorage.getItem('user')
    return stored ? JSON.parse(stored) : null
  })
  const [token, setToken] = useState<string | null>(() => localStorage.getItem('token'))
  ```
- `login(token, user)`: setToken + setUser + localStorage.setItem for both
- `logout()`: setToken(null) + setUser(null) + localStorage.removeItem for both
- `updateToken(newToken)`: setToken(newToken) + localStorage.setItem('token', newToken) — does NOT change user
- `isAuthenticated`: derived as `token !== null && user !== null`
- Export `AuthProvider` component wrapping children with `AuthContext.Provider`
- Export `useAuth` hook that throws if used outside provider
- Export `AuthContext` (for the api.ts interceptor to access outside React tree — use a ref pattern):
  - Create a module-level `authContextRef: { current: AuthContextValue | null } = { current: null }`
  - In AuthProvider, set `authContextRef.current = contextValue` in a useEffect
  - Export `authContextRef` for use by api.ts interceptor
  - This avoids the problem of api.ts running outside React component tree

Note: The `window.location.href` redirect in api.ts does NOT use React Router navigate() because the interceptor runs outside component tree. This is the correct approach per research.

Replace the contents of `frontend/src/services/api.ts` with:

```typescript
import axios from 'axios'
import { authContextRef } from '../contexts/AuthContext'

export const api = axios.create({
  baseURL: import.meta.env.VITE_API_URL,
  headers: {
    'Content-Type': 'application/json',
  },
})

// Request interceptor: attach Bearer token from localStorage
api.interceptors.request.use((config) => {
  const token = localStorage.getItem('token')
  if (token) {
    config.headers.Authorization = `Bearer ${token}`
  }
  return config
})

// Response interceptor:
// 1. Pick up X-New-Token header and update token in context + localStorage
// 2. On 401 from non-auth endpoints: clear auth state and redirect to /login
api.interceptors.response.use(
  (response) => {
    // Check for token renewal header
    const newToken = response.headers['x-new-token']
    if (newToken) {
      localStorage.setItem('token', newToken)
      authContextRef.current?.updateToken(newToken)
    }
    return response
  },
  (error) => {
    if (error.response?.status === 401) {
      const url: string = error.config?.url ?? ''
      const isAuthEndpoint =
        url.includes('/auth/login') ||
        url.includes('/auth/forgot-password') ||
        url.includes('/auth/reset-password')

      if (!isAuthEndpoint) {
        // Clear auth state
        localStorage.removeItem('token')
        localStorage.removeItem('user')
        authContextRef.current?.logout()
        // Redirect outside React tree — window.location is correct here
        window.location.href = '/login'
      }
    }
    return Promise.reject(error)
  },
)
```

Note: Reading token from localStorage in the request interceptor (not from Context) avoids stale closure issues. The Context is the source of truth for React components; localStorage is the source of truth for the interceptor.
